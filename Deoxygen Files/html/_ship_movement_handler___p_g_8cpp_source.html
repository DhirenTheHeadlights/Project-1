27void ShipMovementHandler::updateVelocity(const sf::Vector2f& direction, float elapsedTime, const float baseSpeed) {
28 if (isColliding && speed > 0) speed -= 10.f;
29 else if (!dropAnchor) {
30 // Calculate wind effect
31 GlobalWindController& windController = GlobalWindController::getInstance();
32 sf::Vector2f windDirection = normalize(windController.getWindDirection()); // Ensure wind direction is normalized
33 float windEffect = dot(windDirection, direction) * windController.getWindSpeed();
34
35 // Gradually increase the speed to the base speed, multiplied by the wind effect
36 const float acceleration = std::max(1.f, 1.f * windEffect); // The acceleration factor
37 if (speed < (baseSpeed + windEffect)) speed += acceleration * elapsedTime;
38 else if (speed > (baseSpeed + windEffect)) speed -= acceleration * elapsedTime;
39
40 velocity = sf::Vector2f(direction.x * speed, direction.y * speed);
41
42 // Set the speed immediately before the anchor is dropped
43 speedBeforeAnchorDrop = speed;
44

46 // be set to true again if the anchor is dropped.
47 anchorPushBack = false;

52 const float deceleration = 20.f; // The deceleration factor, proportional to the speed
53
54 // Here, we are going to specify a speed value that will be subtracted from the speed when
55 // the speed goes to 0. This will create a "backwards" effect as the negative speed will
56 // slowly come back up to 0. This is to simulate the ship pulling back from the anchor.
57 // Anchor will pull back more if the ship is moving faster.
58 const float speedMin = -0.2f * speedBeforeAnchorDrop;
59
60 // The reacceleration factor, also proportional to the speed before anchor drop
61 const float reacceleration = 0.2f * speedBeforeAnchorDrop;
62

64
