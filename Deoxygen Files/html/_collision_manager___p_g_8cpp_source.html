147bool CollisionManager::shipCollisionTest(Ship* ship1, Ship* ship2) {
148 // Check for global bounds intersection as a preliminary fast reject
149 if (!ship1->getSprite().getGlobalBounds().intersects(ship2->getSprite().getGlobalBounds())) return false;
150
151 // Get sprite references for ease of use
152 const sf::Sprite& sprite1 = ship1->getSprite();
153 const sf::Sprite& sprite2 = ship2->getSprite();
154
155 // Calculate the positions of the ships' centers
156 sf::Vector2f center1 = sprite1.getPosition();
157 sf::Vector2f center2 = sprite2.getPosition();
158
159 // Calculate the semi-major (a) and semi-minor (b) axes for both ships
160 // Assuming the ships' bounding box width is the semi-major axis and height is the semi-minor axis
161 float a1 = sprite1.getGlobalBounds().width / 2.0f;
162 float b1 = sprite1.getGlobalBounds().height / 2.0f;
163 float a2 = sprite2.getGlobalBounds().width / 2.0f;
164 float b2 = sprite2.getGlobalBounds().height / 2.0f;
165
166 // Calculate the distance between the centers of the two ships
167 float distance = sqrt((center2.x - center1.x) * (center2.x - center1.x) + (center2.y - center1.y) * (center2.y - center1.y));
168
169 // Use an adjusted sum of semi-major axes as a simple overlap check
170 // This factor can be tuned based on ship shapes/collision "sensitivity"
171 float collisionDistance = (a1 + a2) * 0.75f;
172
173 // Check if the distance is less than the adjusted sum of semi-major axes
174 if (distance < collisionDistance) {
175 return true; // Collision detected
176 }
177
178 return false; // No collision detected
