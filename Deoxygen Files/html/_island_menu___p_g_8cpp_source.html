78void IslandMenu::addMarketInteractables() {
79 // Add every item from the market to the menu along
80 // with a buy and sell button and the price of the item
81 for (auto& item : market) {
82 // Create the buy button
83 std::function<void()> buyItem = [this, &item]() {
84 if (ship->getInventoryHandler().getGold() >= item.price && item.amount > 0) {
85 // Value that determines how many are bought each second when the button is held down
86 int buyRate = 1;
87
88 // Attempt to find the item in the ship's inventory
89 auto& inventory = ship->getInventoryHandler().getInventory();
90 auto it = std::find_if(inventory.begin(), inventory.end(), [&item](const ShopItem& inventoryItem) {
91 return inventoryItem.name == item.name;
92 });
93
94 if (it != inventory.end()) {
95 // If the item exists, update its amount
96 it->amount += buyRate;
97 }
98 else {
99 // Otherwise, add a new item with amount 1
100 ShopItem newItem = item;
101 newItem.amount = 1;
102 inventory.push_back(newItem);
103 }
104
105 // Update gold
106 ship->getInventoryHandler().removeGold(item.price);
107
108 // Update merchant's gold and item quantity if applicable
109 item.amount -= buyRate;
110 gold += item.price;
111 }
112 };
113 std::unique_ptr<Button> buyButton = std::make_unique<Button>(buyItem);
114 sf::Text buyButtonText = sf::Text("Buy", font, interactableTextSizeSmall);
115 buyButton->createInteractable(GlobalTextureHandler::getInstance().getMarketTextures().getBuySell(), buyButtonText);
116 buyButton->setHoldDown(true);
117 buyButtons.push_back(std::move(buyButton));
118
119 // Create the sell button
120 std::function<void()> sellItem = [this, &item]() {
121 auto& inventory = ship->getInventoryHandler().getInventory();
122 auto it = std::find_if(inventory.begin(), inventory.end(), [&item](const ShopItem& inventoryItem) {
123 return inventoryItem.name == item.name;
124 });
125
126 if (it != inventory.end() && it->amount > 0) {
127 // If the item exists and has a positive amount, decrease the amount
128 it->amount -= 1;
129
130 // Remove the item from inventory if amount becomes 0
131 if (it->amount == 0) {
132 inventory.erase(it);
133 }
134
135 // Update gold
136 ship->getInventoryHandler().addGold(item.price);
137
138 // Update merchant's gold and item quantity if applicable
139 item.amount += 1;
140 gold -= item.price;
141 }
142 };
143
144 std::unique_ptr<Button> sellButton = std::make_unique<Button>(sellItem);

279void IslandMenu::updateMarket() {
280 // Synchronize the local ship inventory for display purposes
281 auto& currentInventory = ship->getInventoryHandler().getInventory();
282
283 // Update or add new boxes based on the current ship inventory
284 for (size_t i = 0; i < market.size(); ++i) {
285 if (i < shipInventoryInteractable.size()) {
286 // Update existing box
287 auto it = std::find_if(currentInventory.begin(), currentInventory.end(), [&](const ShopItem& item) {
288 return item.name == market[i].name;
289 });
290 shipInventoryInteractable[i]->getText().setString(it != currentInventory.end() ? std::to_string(it->amount) : "Empty");
291 }
292 else {
293 // Add new box if necessary
294 std::unique_ptr<TextDisplayBox> itemDisplay = std::make_unique<TextDisplayBox>();
295 sf::Text itemText = sf::Text(std::to_string(market[i].amount), font, interactableTextSizeSmall);
296 itemDisplay->createInteractable(GlobalTextureHandler::getInstance().getMarketTextures().getMarketFarLeftRight(), itemText);
297 auto it = std::find_if(currentInventory.begin(), currentInventory.end(), [&](const ShopItem& item) {
298 return item.name == market[i].name;
299 });
300 shipInventoryInteractable.push_back(std::move(itemDisplay));
301 }
302 }
303
304 // Remove excess boxes if the current inventory is smaller than the number of boxes
305 if (market.size() < shipInventoryInteractable.size()) {
306 shipInventoryInteractable.resize(market.size());
307 }
308
309 // Update or add new boxes based on the current item amount in the market
310 for (size_t i = 0; i < market.size(); ++i) {
311 if (i < marketInventory.size()) {
312 // Update existing box
313 marketInventory[i]->getText().setString(std::to_string(market[i].amount));
314 }
315 else {
316 // Add new box if necessary
317 std::unique_ptr<TextDisplayBox> marketInventoryAmount = std::make_unique<TextDisplayBox>();
318 sf::Text marketInventoryText = sf::Text(market[i].name + " - " + std::to_string(market[i].amount), font, interactableTextSizeSmall);
319 marketInventoryAmount->createInteractable(GlobalTextureHandler::getInstance().getMarketTextures().getMarketFarLeftRight(), marketInventoryText);
320 marketInventory.push_back(std::move(marketInventoryAmount));
321 }
322 }
323
324 // Update gold
325 shipGoldDisplay->getText().setString("Gold: " + floatToString(ship->getInventoryHandler().getGold()));
326 islandGoldDisplay->getText().setString("Gold: " + floatToString(gold));
327}

330void IslandMenu::draw() {
331 // if the player has said no, do not draw the menu
332 if (hasPlayerSaidNo) return;
333
334 // Draw the ship inventory if the player has entered the island
335 if (!addedInteractables) {
336 addShipInventoryInteractables();
337 addInteractablesToMenu();
338 addedInteractables = true;
339 }
340
341 setInteractablePositions();
342
343 if (enteredIsland) {
344 // Draw menu
345 window->draw(menu);
346
347 // Draw the market interactables
348 for (size_t i = 0; i < buyButtons.size(); ++i) {
349 buyButtons[i]->draw();
350 sellButtons[i]->draw();
351 merchandise[i]->draw();
352 shipInventoryInteractable[i]->draw();
353 marketInventory[i]->draw();
354 }
355
356 // Interact with the market
357 for (size_t i = 0; i < buyButtons.size(); ++i) {
358 buyButtons[i]->interact();
359 sellButtons[i]->interact();
360 uiButtons[0]->interact();
361 }
362
363 // Draw the other ui buttons
364 for (auto& button : uiButtons) {
365 button->draw();
366 }
367
368 // Update the market
369 updateMarket();
370
371 // Draw information
372 shipGoldDisplay->draw();
373 shipNameDisplay->draw();
374 marketNameDisplay->draw();
375 islandGoldDisplay->draw();
376
377 // Draw the island name
378 islandNameDisplay->draw();
379 }
380 else {
381 // Draw the inital menu
382 window->draw(initialMenu);
383
384 // Draw the banner
385 window->draw(banner);
386 window->draw(titleText);
387
388 // Draw the inital buttons and interact with them
389 for (auto& interactable : interactables) {
390 interactable->draw();
391 interactable->interact();
392 }
393 }
