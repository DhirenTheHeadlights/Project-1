33// This function uses a poisson disc sampling algorithm to generate a set of points
34// that are at least a certain distance apart from each other. This function should be
35// called once to generate the points.
36std::vector<sf::Vector2f> const Map::getRandomPositions(float minDistance, int numPoints) {
37 const int k = 30; // attempts before giving up on finding a new point
38 const float pi = 3.14159265358979323846f;
39 std::random_device rd;
40 std::mt19937 gen(rd());
41 std::uniform_real_distribution<float> dis(0.0, 1.0);
42
43 std::vector<sf::Vector2f> samplePoints;
44 std::vector<sf::Vector2f> activeList;
45
46 // Initialize with the first point
47 sf::Vector2f initialPoint(dis(gen) * static_cast<float>(len), dis(gen) * static_cast<float>(height));
48 activeList.push_back(initialPoint);
49 samplePoints.push_back(initialPoint);
50
51 while (!activeList.empty() && samplePoints.size() < static_cast<size_t>(numPoints)) {
52 std::uniform_int_distribution<> dist(0, static_cast<int>(activeList.size()) - 1);
53 int randIndex = dist(gen);
54 sf::Vector2f point = activeList[randIndex];
55 bool found = false;
56
57 for (int i = 0; i < k && samplePoints.size() < static_cast<size_t>(numPoints); ++i) {
58 float angle = dis(gen) * 2 * pi;
59 float radius = minDistance * (1 + dis(gen));
60 sf::Vector2f newPoint(point.x + radius * cos(angle), point.y + radius * sin(angle));
61
62 // Check bounds and minimum distance
63 if (newPoint.x >= 0 && newPoint.x <= len && newPoint.y >= 0 && newPoint.y <= height) {
64 bool tooClose = false;
65 for (const sf::Vector2f& existingPoint : samplePoints) {
66 float distance = std::hypot(newPoint.x - existingPoint.x, newPoint.y - existingPoint.y);
67 if (distance < minDistance) {
68 tooClose = true;
69 break;
70 }
71 }
72 if (!tooClose) {
73 activeList.push_back(newPoint);
74 samplePoints.push_back(newPoint);
75 found = true;
76 if (samplePoints.size() >= static_cast<size_t>(numPoints)) {
77 break; // Exit if we have reached the desired number of points
78 }
79 }
80 }
81 }
82
83 if (!found) {
84 // If no point is found after k attempts, remove the current point from the active list
85 activeList.erase(activeList.begin() + randIndex);
86 }
87 }
88
89 return samplePoints;
90}
