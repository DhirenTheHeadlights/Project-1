151 if (mousePos.x < 0 || mousePos.x > windowSize.x || mousePos.y < 0 || mousePos.y > windowSize.y) {
152 return; // Exit if the mouse is outside the window
153 }
154
155 if (event.type == sf::Event::MouseButtonPressed && event.mouseButton.button == sf::Mouse::Left) {
156 if (!mousePressed && currentGameState == GameState::GameLoop) {
157 Target* target = hashmap.getTargetAtCollision(mousePos, size);
158 if (target != nullptr) {
159 target->deActivate();
160 target->~Target();
161 score += 100; // Increase score for hitting a target

167 mousePressed = true;
