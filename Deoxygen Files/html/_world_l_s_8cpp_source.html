138std::vector<sf::Vector2f> World::calculateShadow(const sf::RectangleShape& box, const sf::Vector2f& lightPos) {
139 std::vector<sf::Vector2f> cornerPoints;
140 std::vector<sf::Vector2f> castDirections;
141
142 // Get the position of the box and its size
143 sf::Vector2f boxPos = box.getPosition();
144 sf::Vector2f boxSize = box.getSize();
145
146 // Calculate the corners of the box
147 std::vector<sf::Vector2f> corners = {
148 boxPos,
149 boxPos + sf::Vector2f(boxSize.x, 0),
150 boxPos + boxSize,
151 boxPos + sf::Vector2f(0, boxSize.y)
152 };
153
154
155 // Calculate shadow points for each corner
156 for (const auto& corner : corners) {
157 sf::Vector2f dir = corner - lightPos;
158 float length = std::sqrt(dir.x * dir.x + dir.y * dir.y);
159 // Make the shadow length proportional to the distance from the light
160
161 if (!box.getGlobalBounds().contains(corner + (dir / length))) {
162 dir.x *= 100; // Length of the shadow, should be long enough to go off-screen
163 dir.y *= 100;
164 cornerPoints.push_back(corner);
165 castDirections.push_back(dir);
166 std::cout << dir.x << " " << dir.y << std::endl;
167 }
168
169 }
170
171
172 std::vector<sf::Vector2f> shadowCasts{};
173 for (int i{ 0 }; i < cornerPoints.size(); i++) {
174 shadowCasts.push_back(cornerPoints[i] + castDirections[i]);
175 }
176 std::vector<sf::Vector2f> shadowPoints{};
177 for (const auto& corner : cornerPoints) {
178 shadowPoints.push_back(corner);
179 }
180 for (const auto& shadowVertex : shadowCasts) {
181 shadowPoints.push_back(shadowVertex);
182 }
183
184
185 return shadowPoints;
