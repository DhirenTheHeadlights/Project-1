267void OptionsMenu::drawTabInteractables(std::vector<std::unique_ptr<Interactable>>& tabInteractables) {
268 // Draw the interactables in reverse order so that the top interactable is draw last
269 // This means that the drop down menu will be drawn on top of the other interactables and will
270 // be visible
271 for (auto it = tabInteractables.rbegin(); it != tabInteractables.rend(); ++it) {
272 (*it)->getText().setFillColor(sf::Color::Black);
273 (*it)->draw();
